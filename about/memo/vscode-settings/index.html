VSCodeの設定

Command + Shift + P から settings.json を開きます

拡張機能を入れてないと、設定に書くだけでは意味がないものも含んでいます 例えば Flutter がインストールされているパスはそれぞれの PC によって変えてみてください 使えないものは無視されるのでエラーにはなりません


{
  // テーマ・カラー
  "window.autoDetectColorScheme": true,
  "workbench.preferredLightColorTheme": "Material Theme Lighter",
  "workbench.preferredDarkColorTheme": "GitHub Dark Dimmed",
  "workbench.colorTheme": "Material Theme Lighter",
  "workbench.iconTheme": "material-icon-theme",
  "editor.minimap.enabled": true,
  "workbench.colorCustomizations": {
    "editorError.background": "#ee5ac4",
    "editorError.foreground": "#ffffff",
    "list.errorForeground": "#ff6fd6",
    "minimap.errorHighlight": "#ff6fd6",
    "scrollbarSlider.background": "#7b75ff"
  },

  // サイズ
  "window.zoomLevel": 1,
  "workbench.tree.indent": 20,
  "editor.tabSize": 7,
  "editor.fontSize": 15,
  "editor.detectIndentation": true,
  "editor.suggestFontSize": 8,
  "terminal.integrated.fontSize": 14,
  "[Log]": {
    "editor.fontSize": 12
  },

  // 起動・準備
  "security.workspace.trust.untrustedFiles": "open",
  "window.restoreWindows": "none",
  "workbench.startupEditor": "none",

  // フォーマット・保存
  "editor.defaultFormatter": "esbenp.prettier-vscode",
  "editor.formatOnSave": false,

  // 余計なポップアップも表示にする エラー説明だけ見たいのにそうはできない様子
  // https://github.com/microsoft/vscode/issues/15667
  "editor.hover.enabled": true,

  // Web
  "[js]": {
    "editor.formatOnSave": true
  },
  "[javascript]": {
    "editor.formatOnSave": true
  },
  "[json]": {
    "editor.defaultFormatter": "vscode.json-language-features"
  },
  "[jsonc]": {
    "editor.formatOnSave": true
  },
  "[html]": {
    "editor.defaultFormatter": "vscode.html-language-features"
  },

  // Unity
  "[csharp]": {
    "editor.defaultFormatter": "ms-dotnettools.csharp",
    "editor.codeLens": false // C#の refarence を非表示
  },
  "omnisharp.useEditorFormattingSettings": true,

  // Markdown
  "markdown.preview.fontSize": 12,
  "markdown.preview.breaks": false,
  "markdown.extension.orderedList.marker": "one",
  "markdown.extension.orderedList.autoRenumber": false,
  "markdown.extension.syntax.decorations": false,
  "markdown.extension.tableFormatter.enabled": false,
  "[markdown]": {
    "editor.formatOnSave": true
  },
  "markdown-pdf.displayHeaderFooter": false,
  "markdown-pdf.margin.bottom": "0.4cm",
  "markdown-pdf.margin.left": "0.5cm",
  "markdown-pdf.margin.right": "0.5cm",
  "markdown-pdf.margin.top": "0.5cm",

  // その他
  "workbench.editorAssociations": {
    "git-rebase-todo": "default"
  }
}
